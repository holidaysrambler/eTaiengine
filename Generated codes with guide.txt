```typescript
import { Website, UserQuery, MatchResult } from '@/types';
import { websiteDatabase } from '@/data/website-database';

export class RulesEngine {
  private static instance: RulesEngine;
  private rules: MatchingRule[] = [];

  private constructor() {
    this.initializeRules();
  }

  static getInstance(): RulesEngine {
    if (!RulesEngine.instance) {
      RulesEngine.instance = new RulesEngine();
    }
    return RulesEngine.instance;
  }

  private initializeRules() {
    this.rules = [
      // Category-based rules
      {
        name: 'category-match',
        weight: 0.3,
        evaluate: (website: Website, query: UserQuery) => {
          if (!query.category) return 0;
          return website.category.toLowerCase() === query.category.toLowerCase() ? 1 : 0;
        }
      },
      
      // Feature-based rules
      {
        name: 'feature-match',
        weight: 0.25,
        evaluate: (website: Website, query: UserQuery) => {
          if (!query.features?.length) return 0;
          const matchedFeatures = query.features.filter(feature => 
            website.features.some(f => f.toLowerCase().includes(feature.toLowerCase()))
          );
          return matchedFeatures.length / query.features.length;
        }
      },
      
      // Style-based rules
      {
        name: 'style-match',
        weight: 0.2,
        evaluate: (website: Website, query: UserQuery) => {
          if (!query.style) return 0;
          return website.designStyle.some(style => 
            style.toLowerCase().includes(query.style!.toLowerCase())
          ) ? 1 : 0;
        }
      },
      
      // Complexity-based rules
      {
        name: 'complexity-match',
        weight: 0.15,
        evaluate: (website: Website, query: UserQuery) => {
          if (!query.complexity) return 0;
          return website.complexity === query.complexity ? 1 : 0;
        }
      },
      
      // Keyword-based rules
      {
        name: 'keyword-match',
        weight: 0.1,
        evaluate: (website: Website, query: UserQuery) => {
          const keywords = query.input.toLowerCase().split(/\s+/);
          const searchableText = [
            website.name,
            website.description,
            ...website.tags,
            ...website.features,
            ...website.designStyle,
            website.category
          ].join(' ').toLowerCase();
          
          const matchedKeywords = keywords.filter(keyword => 
            searchableText.includes(keyword)
          );
          return matchedKeywords.length / keywords.length;
        }
      }
    ];
  }

  public findMatches(query: UserQuery, limit: number = 10): MatchResult[] {
    const results: MatchResult[] = [];

    for (const website of websiteDatabase) {
      let totalScore = 0;
      const matchedTerms: string[] = [];

      for (const rule of this.rules) {
        const score = rule.evaluate(website, query);
        totalScore += score * rule.weight;
        
        if (score > 0) {
          matchedTerms.push(rule.name);
        }
      }

      if (totalScore > 0) {
        results.push({
          website,
          matchScore: totalScore,
          matchedTerms,
          reason: this.generateMatchReason(website, query, matchedTerms)
        });
      }
    }

    // Sort by match score and return top results
    return results
      .sort((a, b) => b.matchScore - a.matchScore)
      .slice(0, limit);
  }

  private generateMatchReason(website: Website, query: UserQuery, matchedTerms: string[]): string {
    const reasons: string[] = [];
    
    if (matchedTerms.includes('category-match')) {
      reasons.push(`Matches your ${query.category} category`);
    }
    
    if (matchedTerms.includes('feature-match')) {
      reasons.push(`Has the features you're looking for`);
    }
    
    if (matchedTerms.includes('style-match')) {
      reasons.push(`Fits your ${query.style} design style`);
    }
    
    if (matchedTerms.includes('keyword-match')) {
      reasons.push(`Matches keywords from your description`);
    }
    
    return reasons.join(', ');
  }
}

interface MatchingRule {
  name: string;
  weight: number;
  evaluate: (website: Website, query: UserQuery) => number;
}
```

```typescript
import { UserQuery } from '@/types';
import { RulesEngine } from '@/lib/rules-engine';

interface SearchEngineProps {
  onResultsFound: (results: any[]) => void;
  onNoMatches: () => void;
}

export function SearchEngine({ onResultsFound, onNoMatches }: SearchEngineProps) {
  const [searchInput, setSearchInput] = useState('');
  const [selectedCategory, setSelectedCategory] = useState('All');
  const [selectedStyle, setSelectedStyle] = useState('');
  const [selectedComplexity, setSelectedComplexity] = useState('');
  const [features, setFeatures] = useState<string[]>([]);

  const rulesEngine = useMemo(() => RulesEngine.getInstance(), []);

  const handleSearch = () => {
    const query: UserQuery = {
      input: searchInput,
      ...(selectedCategory !== 'All' && { category: selectedCategory }),
      ...(selectedStyle && { style: selectedStyle }),
      ...(selectedComplexity && { complexity: selectedComplexity }),
      ...(features.length > 0 && { features })
    };

    const results = rulesEngine.findMatches(query);
    
    if (results.length > 0) {
      onResultsFound(results);
    } else {
      onNoMatches();
    }
  };

  const availableFeatures = useMemo(() => {
    const allFeatures = new Set<string>();
    websiteDatabase.forEach(site => {
      site.features.forEach(feature => allFeatures.add(feature));
    });
    return Array.from(allFeatures);

  return (
```

```typescript
'use client';

import { useState } from 'react';
import { SearchEngine } from '@/components/matching/SearchEngine';
import { ResultsGrid } from '@/components/matching/ResultsGrid';
import { BuilderInterface } from '@/components/builder/BuilderInterface';
import { MatchResult, BuilderComponent, StyleConfig } from '@/types';

type AppMode = 'search' | 'results' | 'builder';

export default function Home() {
  const [mode, setMode] = useState<AppMode>('search');
  const [searchResults, setSearchResults] = useState<MatchResult[]>([]);
  const [builderComponents, setBuilderComponents] = useState<BuilderComponent[]>([]);
  const [builderStyles, setBuilderStyles] = useState<StyleConfig>({
    colors: {
      primary: '#2563eb',
      secondary: '#64748b',
      accent: '#f59e0b',
      background: '#ffffff',
      text: '#1e293b'
    },
    typography: {
      fontFamily: 'Inter, sans-serif',
      fontSize: '16px',
      fontWeight: 'normal'
    },
    spacing: 'comfortable',
    borderRadius: '8px'
  });

  const handleResultsFound = (results: MatchResult[]) => {
    setSearchResults(results);
    setMode('results');
  };

  const handleNoMatches = () => {
    setMode('builder');
  };

  const handleStartBuilding = () => {
    setMode('builder');
  };

  const handleBackToSearch = () => {
    setMode('search');
    setSearchResults([]);
  };

  const handleAddComponent = (component: BuilderComponent) => {
    setBuilderComponents([...builderComponents, component]);
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-purple-50 to-blue-50">
      {/* Header */}
      <header className="bg-white shadow-sm border-b">
        <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
          <div className="flex justify-between items-center py-4">
            <div className="flex items-center space-x-3">
              <div className="w-8 h-8 bg-blue-600 rounded-lg"></div>
              <h1 className="text-xl font-bold text-gray-900">UI Matcher & Builder</h1>
            </div>
            
            <div className="flex items-center space-x-4">
              {mode !== 'search' && (
                <button
                  onClick={handleBackToSearch}
                  className="px-4 py-2 text-sm text-gray-600 hover:text-gray-900"
                >
                  â† Back to Search
                </button>
              )}
              
              <div className="flex items-center space-x-2 text-sm text-gray-500">
                <div className={`w-2 h-2 rounded-full ${
                  mode === 'search' ? 'bg-green-500' : 'bg-gray-300'
                }`}></div>
                <span>Search</span>
                
                <div className={`w-2 h-2 rounded-full ${
                  mode === 'results' ? 'bg-green-500' : 'bg-gray-300'
                }`}></div>
                <span>Results</span>
                
                <div className={`w-2 h-2 rounded-full ${
                  mode === 'builder' ? 'bg-green-500' : 'bg-gray-300'
                }`}></div>
                <span>Builder</span>
              </div>
            </div>
          </div>
        </div>
      </header>

      {/* Main Content */}
      <main>
        {mode === 'search' && (
          <div className="py-12">
            <div className="text-center mb-12">
              <h2 className="text-4xl font-bold text-gray-900 mb-4">
                Find Your Perfect Website Design
              </h2>
              <p className="text-lg text-gray-600 max-w-2xl mx-auto">
                Describe what you need, and we'll match you with proven designs or help you build from scratch using our custom rules engine.
              </p>
            </div>
            <SearchEngine 
              onResultsFound={handleResultsFound}
              onNoMatches={handleNoMatches}
            />
          </div>
        )}

        {mode === 'results' && (
          <div className="py-8">
            <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
              <div className="text-center mb-8">
                <h2 className="text-3xl font-bold text-gray-900 mb-2">
                  We Found {searchResults.length} Matches
                </h2>
                <p className="text-gray-600">
                  Based on your requirements and our custom matching rules
                </p>
              </div>
              
              <ResultsGrid 
                results={searchResults} 
                onStartBuilding={handleStartBuilding}
              />
            </div>
          </div>
        )}

        {mode === 'builder' && (
          <BuilderInterface
            components={builderComponents}
            styles={builderStyles}
            onAddComponent={handleAddComponent}
            onStylesChange={setBuilderStyles}
          />
        )}
      </main>
    </div>
  );
}
```
